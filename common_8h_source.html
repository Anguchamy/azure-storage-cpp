303  const result_type& operator*() const
304  {
305  if (passed_the_end())

318  result_iterator& operator++()
319  {
320  // preincrement
321  // - if the iterator has passed the end, do nothing
322  // - if any exception is thrown within the method, do nothing
323  if (!passed_the_end())
324  {
325  if (m_segment_index + 1 == m_result_segment.results().size() && !m_result_segment.continuation_token().empty())
326  {
327  fetch_next_segment();
328  }
329  else
330  {
331  m_segment_index++;
332  }
333 
334  m_returned_results++;
335  }
336 
337  return (*this);
338  }

348  bool operator==(const result_iterator& rhs) const
349  {
350  // test for iterator equality
351  // - comparison of two past-the-end iterators returns true
352  // - comparison of a non-end iterator and a past-the-end iterators returns false
353  // - comparison of two non-end iterators returns false
354  return passed_the_end() && rhs.passed_the_end();
355  }

364  void fetch_first_segment()
365  {
366  if (nullptr != m_result_generator)
367  {
368  m_result_segment = m_result_generator(continuation_token(), get_remaining_results_num());
369  m_segment_index = 0;
370 
371  if (m_result_segment.results().empty())
372  {
373  // continue if returned result segment is empty
374  fetch_next_segment();
375  }
376  }
377  }
378 
379  void fetch_next_segment()
380  {
381  if (nullptr != m_result_generator && !m_result_segment.continuation_token().empty())
382  {
383  auto tmp_segment = m_result_generator(m_result_segment.continuation_token(), get_remaining_results_num());
384  while (tmp_segment.results().empty() && !tmp_segment.continuation_token().empty())
385  {
386  tmp_segment = m_result_generator(tmp_segment.continuation_token(), get_remaining_results_num());
387  }
388 
389  m_result_segment = std::move(tmp_segment);
390  m_segment_index = 0;
391  }
392  }
393 
394  size_t get_remaining_results_num() const
395  {
396  if (m_max_results == 0)
397  {
398  // 0 indicates to return maximum possible number of results
399  return m_max_results_per_segment;
400  }
401 
402  return (size_t)std::min(m_max_results - m_returned_results, (utility::size64_t)m_max_results_per_segment);
403  }
404 
405  bool passed_the_end() const
406  {
407  return (m_segment_index == m_result_segment.results().size() && m_result_segment.continuation_token().empty()) ||
408  (m_max_results > 0 && m_returned_results >= m_max_results);
409  }
